109    def _callback_server(self) -> dict:
110        """Run a callback server to wait for the API authorisation response.
111
112        Returns:
113            dict: Response parameters.
114        """
115
116        done = False
117        params = None
118
119        class Handler(http.server.BaseHTTPRequestHandler):
120
121            def do_GET(self):
122                nonlocal done, params
123
124                if self.path.startswith("/?code="):  # Successful callback
125
126                    self.send_response(200, "Ok")
127                    self.end_headers()
128
129                    # Create a dict of the params, should be code and state
130                    params = TredictPy.params_from_path(self.path)
131
132                    self.wfile.write("Authorisation complete!".encode("utf-8"))
133                    done = True
134
135                elif self.path.startswith("/?error="):  # Error callback
136
137                    self.send_response(200, "Ok")
138                    self.end_headers()
139
140                    # Create a dict of the params, should be code and state
141                    params = TredictPy.params_from_path(self.path)
142
143                    self.wfile.write("Authorisation failed!".encode("utf-8"))
144                    done = True
145
146                elif self.path.startswith("/favicon.ico"):  # Add favicon at some point
147                    self.send_response(404, "Not Found")
148                    self.end_headers()
149                elif self.path.startswith("/privacy"):  # Will add a privacy policy
150                    self.send_response(204, "No Content")
151                    self.end_headers()
152                else:  # A page that does not exist was requested
153                    self.send_response(404, "Not Found")
154                    self.end_headers()
155
156        with TCPServer(("localhost", 8080), Handler) as httpd:
157            print("Callback server started...")
158            while not done:
159                httpd.handle_request()
160            print("Callback server stopped.")
161
162        return params

174    def request_auth_code(self, headless: bool = False) -> None:
175        """Request an authorisation code.
176
177        Provides a link to authorise on Tredict then either starts a callback server to capture the response or awaits
178        the URL to be entered for headless.
179
180        Args:
181            headless (bool, optional): Run in headless mode. Defaults to False.
182
183        Raises:
184            APIException: If the returned and supplied sates do not match or the authorisation failed.
185        """
186
187        user_uuid = str(uuid.uuid4())
188
189        print(
190            f"Open this URL to authorise: {self._config['auth_url']}?client_id={self._config['client_id']}&state={user_uuid}"
191        )
192
193        # Start the callback server or go headless
194        params = self._callback_headless() if headless else self._callback_server()
195
196        if "code" in params.keys() and params["state"] == user_uuid:
197            print("Authorisation complete!")
198            self.save_config({"auth_code": params})
199        elif "code" in params.keys() and params["state"] != user_uuid:
200            raise APIException(
201                f"Authorisation failed! Returned state does not match supplied state."
202            )
203        else:  # If code is not in the keys authorisation failed
204            raise APIException(
205                f"Authorisation failed!\nCallback response:\n{json.dumps(params, indent=4)}"

208    def request_user_access_token(self, refresh: bool = False) -> None:
209        """Request a user access token using either an authorisation code or a refresh token.
210
211        Args:
212            refresh (bool, optional): Set to True to use a long lived refresh token. Defaults to False.
213
214        Raises:
215            APIException: If there was an error requesting the user access token.
216        """
217
218        if not refresh and (
219            self._config["auth_code"] is None or "auth_code" not in self._config.keys()
220        ):
221            raise APIException("You must request an authorisation code first.")
222
223        if refresh and (
224            self._config["user_access_token"] is None
225            or "user_access_token" not in self._config.keys()
226        ):
227            raise APIException(
228                "You must request a user access token with an authorisation code before you can refresh."
229            )
230
231        headers = {
232            "content-type": "application/x-www-form-urlencoded",
233            "accept": "application/json;charset=UTF-8",
234        }
235
236        data = {
237            "grant_type": "refresh_token" if refresh else "authorization_code",
238            "code": None if refresh else self._config["auth_code"]["code"],
239            "refresh_token": (
240                self._config["user_access_token"]["refresh_token"] if refresh else None
241            ),
242        }
243
244        r = requests.post(
245            f"{self._config['token_url']}{self._config['token_append']}",
246            headers=headers,
247            auth=(self._config["client_id"], self._config["client_secret"]),
248            data=data,
249        )
250
251        if r.status_code == 200:
252            print("User access token successfully retrieved!")
253
254            user_access_token = {
255                "user_access_token": r.json()
256                | {"expires_on": int(time.time() + r.json()["expires_in"])}
257            }
258
259            if refresh:
260                user_access_token["user_access_token"].update(
261                    {"refresh_token": data["refresh_token"]}
262                )
263
264            self.save_config(user_access_token)
265        else:
266            raise APIException(
267                f"Retrieving user access token failed error {r.status_code} ({self._config['error_codes'][str(r.status_code)]})."
268            )

300    def _list_endpoint(self, endpoint: str, params: dict) -> list:
301        """Make a request to a list endpoint.
302
303        Handles pagination.
304
305        Args:
306            endpoint (str): Name of the list endpoint.
307            params (dict): Parameters as required by the endpoint.
308
309        Raises:
310            APIException: If the request fails.
311
312        Returns:
313            list: A list of the response pages.
314        """
315
316        if (
317            self._config["user_access_token"] is None
318            or "user_access_token" not in self._config.keys()
319        ):
320            raise APIException("You must request a user access token first.")
321
322        headers = {
323            "authorization": f"bearer {self._config['user_access_token']['access_token']}",
324            "accept": "application/json;charset=UTF-8",
325        }
326
327        pages = []
328        url = f"{self._config['endpoint_base_url']}{endpoint}/{self._config['endpoint_append']}"
329
330        while True:
331
332            r = requests.get(
333                url,
334                headers=headers,
335                params=params,
336            )
337
338            if r.status_code == 200:
339                pages.extend(
340                    r.json()["_embedded"][list(r.json()["_embedded"].keys())[0]]
341                )
342
343                if (
344                    "_links" not in r.json().keys()
345                    or "next" not in r.json()["_links"].keys()
346                ):
347                    break
348                else:
349                    url = r.json()["_links"]["next"]
350                    # Also need to set params to None as next contains params
351                    params = None
352            else:
353                raise APIException(
354                    f"Request to {endpoint} failed error {r.status_code} ({self._config['error_codes'][str(r.status_code)]})."
355                )
356
357        return pages
